Currencyservlet.java import java.io.*; import jakarta.servlet.*; import jakarta.servlet.http.*; public class CurrencyServlet extends HttpServlet { protected void doPost(HttpServletRequest req, HttpServletResponse res) throws ServletException, IOException { double amount = Double.parseDouble( req.getParameter("amount")); String from = req.getParameter("from"); String to = req.getParameter("to"); double converted = 0; // Predefined Conversion Rates if(from.equals("INR") && to.equals("USD")){ converted = amount * 0.012; } else if(from.equals("USD") && to.equals("INR")){ converted = amount * 83; } else if(from.equals("INR") && to.equals("EUR")){ converted = amount * 0.011; } else if(from.equals("EUR") && to.equals("INR")){ converted = amount * 90; } else if(from.equals("USD") && to.equals("EUR")){ converted = amount * 0.92; } else if(from.equals("EUR") && to.equals("USD")){ converted = amount * 1.08; } else{ converted = amount; } req.setAttribute("amount", amount); req.setAttribute("from", from); req.setAttribute("to", to); req.setAttribute("converted", converted); RequestDispatcher rd = req.getRequestDispatcher("result.jsp"); rd.forward(req,res); } } Result.jsp Conversion Result
